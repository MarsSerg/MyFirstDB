  //функция записи и изменения строки
  function readstring(x, y, textcolor, maxlength: integer): string;
  var
    s: string;
    pos: integer;
    key: char;
    flag: boolean;
  begin
    flag := True;
    setlength(s, maxlength);
    pos := 0;
    gotoxy(x, y);
    repeat

      key := readkey;
      if (key <> #8) and (key <> #13) and (pos < maxlength) then
      begin
        Write(key);
        pos := pos + 1;
        s[pos] := key;
      end;
      //writeln(ord(key));
      if (key = #8) and (pos <> 0) then
      begin
        pos := pos - 1;
        gotoxy(x + pos, y);
        Write(' ');
        gotoxy(x + pos, y);
      end;
      if (key = #13) then
        flag := False;
    until flag = False;
    setlength(s, pos);
    readstring := s;
  end;

  //ввод чилса и проверка было ли оно введено верно
  procedure readlnnumber(var n: integer);
  var
    s: string;
    code: integer;
  begin
    readln(s);
    val(s, n, code);

    while code > 0 do
    begin
      writeln('ошибка при вводе числа,введите заново');
      readln(s);
      val(s, n, code);
    end;

  end;
